# Get logs for the consumer by name pattern
aws --region us-east-1 logs describe-log-groups --log-group-name-pattern gkcisco-consumer

# Get log streams for the consumer by name of the log group
aws --region us-east-1 logs describe-log-streams --log-group-name /aws/lambda/gkcisco-consumer

# Get log events in real-time. live tail
aws --region us-east-1 logs start-live-tail --log-group-identifiers "arn:aws:logs:us-east-1:739983581684:log-group:/aws/lambda/gkcisco-consumer"
